  • Load string palette 
-- title:   Load string palette
-- author:  congusbongus, TIC-80 wiki
-- desc:    Load palette from a string
-- license: MIT License
-- script:  moon
-- tags:    draw palette string

loadPalette=(palette)->
  for i=0,15 do
    r=tonumber(string.sub(palette,i*6+1,i*6+2),16)
    g=tonumber(string.sub(palette,i*6+3,i*6+4),16)
    b=tonumber(string.sub(palette,i*6+5,i*6+6),16)
    poke(0x3FC0+(i*3)+0,r)
    poke(0x3FC0+(i*3)+1,g)
    poke(0x3FC0+(i*3)+2,b)

-- Palette strings
-- Go to https://aaronsnoswell.github.io/blog/tic-80-color-palette-tool to make your own palette!
palettes={
	{p:"1a1c2c5d275db13e53ef7d57ffcd75a7f07038b76425717929366f3b5dc941a6f673eff7f4f4f494b0c2566c86333c57",name:"sweetie16"},
	{p:"140c1c44243430346d4e4a4f854c30346524d04648757161597dced27d2c8595a16daa2cd2aa996dc2cadad45edeeed6",name:"classic tic80"},
	{p:"0000001d2b537e255383769cab5236008751ff004d5f574fff77a8ffa300c2c3c700e436ffccaa29adffffec27fff1e8",name:"pico8"},
	{p:"000000111111222222333333444444555555666666777777888888999999aaaaaabbbbbbccccccddddddeeeeeeffffff",name:"grayscale"},
}
palettei=1

t=0

export TIC=->
	-- Cycle palettes
	if btnp(4) or btnp(5)
		palettei+=1
		if palettei>#palettes
			palettei=1
		loadPalette(palettes[palettei].p)

	cls 13
	-- Draw palette squares
	for i=0,7
		rect(10+i*10,10,10,10,i)
	for i=8,15
		rect(10+(i-8)*10,20,10,10,i)
	spr 1+(t%60)//30*2,96,24,14,3,0,0,2,2
	print "Palette: #{palettes[palettei].name}",24,84
	print "Press button to switch palettes",24,96
	t+=1
