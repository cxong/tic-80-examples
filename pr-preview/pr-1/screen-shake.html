  • Screen shake 
-- title:   Screen shake
-- author:  congusbongus, Vadim
-- desc:    Shake the screen
-- license: MIT License
-- script:  moon
-- tags:    draw shake effect

t=0
x=96
y=24
shake=0
d=4

export TIC=->
	if btn 0
		y-=1
	if btn 1
		y+=1
	if btn 2
		x-=1
	if btn 3
		x+=1
	if btnp(4) or btnp(5)
		shake=30
	if shake>0
		poke(0x3FF9,math.random(-d,d))
		poke(0x3FF9+1,math.random(-d,d))
		shake-=1
		if shake==0
			memset(0x3FF9,0,2)

	cls 13
	spr 1+(t%60)//30*2,x,y,14,3,0,0,2,2
	print("PRESS A/B TO SHAKE!",54,64)
	t+=1
