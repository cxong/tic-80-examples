-- title:   Sprite rotate
-- author:  congusbongus
-- desc:    How to rotate a sprite
-- license: MIT License
-- script:  moon
-- tags:    draw rotate sprite

t=0
x=24
y=48

export TIC=->
	cls 13

	rotation=0
	spr 1+(t%60)//30*2,x,y,14,2,0,rotation,2,2
	print "r=#{rotation}",x,y-10

	rotation=1
	spr 1+(t%60)//30*2,x+48,y,14,2,0,rotation,2,2
	print "r=#{rotation}",x+48,y-10

	rotation=2
	spr 1+(t%60)//30*2,x+96,y,14,2,0,rotation,2,2
	print "r=#{rotation}",x+96,y-10

	rotation=3
	spr 1+(t%60)//30*2,x+144,y,14,2,0,rotation,2,2
	print "r=#{rotation}",x+144,y-10

	t+=1
