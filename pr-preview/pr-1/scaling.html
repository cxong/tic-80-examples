-- title:   Sprite scaling
-- author:  congusbongus
-- desc:    How to scale a sprite
-- license: MIT License
-- script:  moon
-- tags:    draw scale sprite

t=0
x=24
y=24

export TIC=->
	cls 13

	scale=1
	spr 1+(t%60)//30*2,x,y,14,scale,0,0,2,2
	print "#{scale}x",x,y-10

	scale=2
	spr 1+(t%60)//30*2,x+24,y,14,scale,0,0,2,2
	print "#{scale}x",x+24,y-10

	scale=3
	spr 1+(t%60)//30*2,x+72,y,14,scale,0,0,2,2
	print "#{scale}x",x+72,y-10

	scale=4
	spr 1+(t%60)//30*2,x+128,y,14,scale,0,0,2,2
	print "#{scale}x",x+128,y-10

	t+=1
