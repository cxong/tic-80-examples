-- title:   Speed Lines
-- author:  congusbongus
-- desc:    Draw animated anime speed lines, moving across or focused on a point
-- license: CC0
-- script:  moon
-- tags:    anime draw animated lines speed

WIDTH=240
HEIGHT=136

-- Util (vector) functions
distance=(x1,y1,x2,y2)->
	dx=x1-x2
	dy=y1-y2
	return math.sqrt(dx*dx+dy*dy)

norm=(x,y)->
	d=distance(x,y,0,0)
	if d==0
		return 1,0
	return x/d,y/d

-- Speedline: one line that moves from right to left
class Speedline
	new:(y,l,c)=>
		@y=y
		@l=l
		@x=WIDTH
		@c=c

	update:(dx)=>
		@x+=dx

	alive:=>return @x+@l>0

	draw:=>
        -- draw two triangles side by side connected by an edge (like this: <|>)
		tri(@x,@y,@x+@l,@y+1,@x+@l,@y-1,@c)
		tri(@x+@l*2,@y,@x+@l,@y+1,@x+@l,@y-1,@c)

-- Constantly generate speedlines across the height of the screen
class Speedlines
	new:=>
		@lines={}

	update:=>
		for i=1,2
			table.insert(@lines,Speedline(math.random(HEIGHT),WIDTH*2+math.random(50),math.random(4)))
		for l in *@lines
			l\update(-50)
		@lines=[l for l in *@lines when l\alive!]

	draw:=>
		cls(15)
		dy=15
		for i=1,4
			rect(0,dy*i,WIDTH,HEIGHT-dy*i*2,i)
		for l in *@lines
			l\draw!

-- A focus line starts at the focus point and stretches/moves outwards
class Focusline
	new:(x,y,dx,dy,c,l)=>
		@x1,@y1=x,y
		@dx1,@dy1=dx,dy
		px,py=dy,-dx
		px/=50
		py/=50
		@x2,@y2=x,y
		@dx2,@dy2=dx+px,dy+py
		@count=0
		@c=c
		@sx1,@sy1=x,y
		@sx2,@sy2=x,y
		@l=l

	update:(d)=>
		@count+=1
		@x1+=@dx1*d
		@y1+=@dy1*d
		@x2+=@dx2*d
		@y2+=@dy2*d
		if @count>@l
			@sx1+=@dx1*d
			@sy1+=@dy1*d
			@sx2+=@dx2*d
			@sy2+=@dy2*d

	alive:=>
		return @sx1>=0 and @sx1<WIDTH and @sy1>=0 and @sy1<HEIGHT

	draw:=>
        -- Draw two triangles connected by an edge
		tri(@sx1,@sy1,@x1,@y1,@x2,@y2,@c)
		tri(@sx1,@sy1,@sx2,@sy2,@x2,@y2,@c)

-- Constantly generate focus lines starting from the focus point and radiating outwards
class Focuslines
	new:(x,y,bgc,c)=>
		@x=x
		@y=y
        -- Focus lines are single color
		@bgc=bgc
		@c=c
		@lines={}

	update:=>
		for i=1,10
			dx,dy=math.random!-0.5,math.random!-0.5
			dx,dy=norm(dx,dy)
			table.insert(@lines,Focusline(@x,@y,dx,dy,@c,10))
		for l in *@lines
			l\update(math.random(40,60))
		@lines=[l for l in *@lines when l\alive!]

	draw:=>
		cls(@bgc)
		for l in *@lines
			l\draw!

state=nil
lines=nil
FOCUS_X=WIDTH/2
FOCUS_Y=HEIGHT/2

export TIC=->
    -- initial state
    if state == nil or btnp(0) or btnp(1) or btnp(2) or btnp(3) or btnp(4) or btnp(5)
        if state != "focus"
            state="focus"
            lines=Focuslines(FOCUS_X,FOCUS_Y,0,13)
        else
            state="speed"
            lines=Speedlines!
    lines\update!

    lines\draw!

    print("state: #{state}",12,HEIGHT-40,11)
    print("press any button to switch",12,HEIGHT-20,11)

    if state == "focus"
        circ(FOCUS_X,FOCUS_Y,5,7)
